```c++
快排模板
void quick_sort(int q[], int l, int r)
{
	if (l >= r)
		return;
	int x = q[l], i = l - 1, j = r + 1;
	while (i < j)
	{
		while (q[++i] < x);
		while (q[--j] > x);
		if (i < j) swap(q[i], q[j]);
	}
	quick_sort(q, l, j);
	quick_sort(q, j + 1, r);
}
```

```c
归并模板
int tmp[1000010];
void merge_sort(int p[], int l, int r)
{
    //对半划分
	if (l >= r)	return;
	int mid = l + r >> 1;
	merge_sort(p, l, mid);
	merge_sort(p, mid + 1, r);

    //合并操作
	int i = l, j = mid + 1, k = 0;
	while (i <= mid && j <= r)
	{
		if (p[i] <= p[j]) tmp[k++] = p[i++];
		else tmp[k++] = p[j++];
	}
	while (i <= mid) tmp[k++] = p[i++];
	while (j <= r) tmp[k++] = p[j++];
	
    //重要
	for (i = l, j = 0; i <= r; ++i, ++j) p[i] = tmp[j];
}
```
