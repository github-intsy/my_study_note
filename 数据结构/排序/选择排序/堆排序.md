    void HeapSort(int* a, int n)//时间复杂度O(N*logN)
    {
        for(int i = (n-1-1)/2; i >= 0; --i)
        {
            AdjustDown(a, n, i);
        }

        int end = n - 1;
        while(end > 0)
        {
            Swap(&a[0], &a[end]);
            AdjustDown(a,end,0);
            --end;
        }
    }


    void AdjustDown(int* a, int n, int root)
    {
        int parent = root;
        int child = parent * 2 + 1;
        while(child < n)
        {
            if(child + 1 < n && a[child + 1] > a[child])
            {
                ++child;
            }
            if(a[child] > a[parent])
            {
                Swap(&a[child], &a[parent]);
                parent = child;
                child = parent * 2 + 1;
            }
            else
            {
                break;
            }
        }
    }
