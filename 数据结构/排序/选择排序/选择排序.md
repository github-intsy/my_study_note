##### 选择一个最大和最小的数分别放到数组的两侧,以此缩小数组范围.
#### 代码:
    void SelectSort(int* a, int a)
    {
        assert(a);
        int begin = 0, end = n - 1;
        while(begin < end)
        {
            //在[begin,end]之间找出最大和最小数的下标
            int mini = maxi = begin;
            for(int i = begin + 1; i <= end; ++i)
            {
                if(a[i] > a[maxi])
                {
                    maxi = i;
                }
                if(a[i] < a[mini])
                {
                    mini = i;
                }
            }

            Swap(&a[begin], &a[mini]);\
            //如果maxi和begin位置重合,则maxi的位置需要修正
            if(begin == maxi)
            {
                maxi = mini;
            }
            Swap(&a[end], &a[maxi]);

            ++begin;
            --end;
        }
    }
    时间复杂度是O(N*N);
